class Essential_Blocks {
	/**
	 * Blocks constructor.

	/**
	 * Register All Blocks
	 */
	public function register_blocks() {

		register_block_type( {{constant}}_DIR . './assets/block/advance-tabs/block.json' );
		register_block_type( {{constant}}_DIR . './assets/block/advance-tab/block.json' );
	}
}
